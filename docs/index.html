调用方式：
  
require([
  'brix/animation',
  'jquery',
  'underscore'
], function(Anim, $, _) {
   //自定义扩展命令: 移除样式
  Anim.extend('removeClass', function(step) {
    var node = step.node
    var className = step.param
    var done = step.done

    node.removeClass(className)
    done() //结束执行done
  })

  var options = {
    el: $('body'),
    owner: {
      testclick: function(e, arg1) {
        $(e.target).html('执行了testclick方法,方法的参数为' + arg1)
      }
    }
  } //配置
  var anim = new Anim(options)

})
内置的命令列表：
  on: //dom事件触发
  class: //添加class
  execute: //调用宿主上的方法
  wait: //延迟等待
  style: //添加内联样式
  call/when: //触发自定义的事件
